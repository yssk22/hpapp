const (
    // Label holds the string label denoting the hpsorthistory type in the database.
    Label = "hp_sort_history"
    // FieldID holds the string denoting the id field in the database.
    FieldID = "id"
    // FieldCreatedAt holds the string denoting the created_at field in the database.
    FieldCreatedAt = "created_at"
    // FieldUpdatedAt holds the string denoting the updated_at field in the database.
    FieldUpdatedAt = "updated_at"
    // FieldSortResult holds the string denoting the sort_result field in the database.
    FieldSortResult = "sort_result"
    // FieldOwnerUserID holds the string denoting the owner_user_id field in the database.
    FieldOwnerUserID = "user_hpsort_history"
    // EdgeOwner holds the string denoting the owner edge name in mutations.
    EdgeOwner = "owner"
    // Table holds the table name of the hpsorthistory in the database.
    Table = "hp_sort_histories"
    // OwnerTable is the table that holds the owner relation/edge.
    OwnerTable = "hp_sort_histories"
    // OwnerInverseTable is the table name for the User entity.
    // It exists in this package in order to avoid circular dependency with the "user" package.
    OwnerInverseTable = "users"
    // OwnerColumn is the table column denoting the owner relation/edge.
    OwnerColumn = "user_hpsort_history"
)

Note that the variables below are initialized by the runtime package on the initialization of the application. Therefore, it should be imported in the main as follows:

var (
    Hooks  [2]ent.Hook
    Policy ent.Policy
)
